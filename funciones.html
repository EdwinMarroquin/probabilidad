getData <- function(file, sheet) {
  proyectRoot = getwd()
  dataUrl <-
    paste("./../data/" ,
          file,
          sep = "",
          collapse = NULL)
  data <- read_excel(gsub("\\\\", "/", dataUrl), sheet = sheet)
  return(data)
}

#' Title
#' @description Busca los valores dulicados de una variable y devuelve un informe con los resultados
findDatabaseDuplicades <- function (label, array) {
  registros <- nrow(array[label])
  registrosUnicos <- nrow(unique(array[label]))
  response <- paste0(
    cat(
      "**VALIDANDO REGISTROS DUPLICADOS DE (",
      toupper(label),
      ")**\n\n",
      sep = "",
      collapse = NULL
    ),
    cat(
      "> El campo **",
      label,
      "** continene <span class='badge bg-primary'>",
      registros,
      "</span> registros",
      "\n>\n",
      sep = "",
      collapse = NULL
    ),
    cat(
      "> El campo **",
      label,
      "** continene <span class='badge bg-primary'>",
      registrosUnicos,
      "</span> registros &uacute;nicos",
      "\n>\n",
      sep = "",
      collapse = NULL
    ),
    cat(
      ifelse((registros == registrosUnicos),
             "> **No existen registros duplicados**",
             "> **Existen registros duplicados**"
      ),
      "\n" ,
      sep = "",
      collapse = NULL
    )
  )
  return(toupper(response))
}

filterDatabase <- function(database, column) {
  result <- database %>% filter(column > 0)

  return(result)
}

getDatabaseClean <- function(file, sheet, variable) {
  data <- getData(file, sheet)

  result <- data %>% filter(data[variable] > 0)

  message <- cat(
    cat(
      "## Proceso de limpieza de datos",
      "\n\n",
      "> RESGISTROS ANTES DE LA LIMPIEZA <span class='badge bg-primary'>",
      nrow(data),
      "</span>",
      "\n",
      ">",
      "\n",
      sep = "",
      collapse = NULL
    ),
    cat(
      "> RESGISTROS LUEGO DE LA LIMPIEZA <span class='badge bg-primary'>",
      nrow(result),
      "</span>",
      "\n",
      sep = "",
      collapse = NULL
    )
  )

  return(list("data" = result, "msg" = message))
}


replaceDataIndex <- function(database, variable) {
  d <- database
  items <- unique(d[[variable]])
  step = 1
  for (item in items) {
    d[[variable]] [d[[variable]] == toString(item)] <- step
    step = step + 1
  }
  return(d)
}

central_moda <- function(database, variable) {
  return(mfv(database[[variable]]))
}

central_media <- function(database, variable) {
  return(round(mean(database[[variable]]), 2))
}

central_mediana <- function(database, variable) {
  return(round(median(database[[variable]]), 2))
}

        col = colR,

      )

tablaContingencia <- function(database, var1, var2) {
  tabla <- table(database[[var1]], database[[var2]])

  print(
    tabla %>% prop.table() %>%  round(4) %>% multiply(100) %>% addmargins()
    %>% kbl(
      caption = str_c(
        "<h4>TABLA DE CONTINGENCIA PARA LAS VARIABLES ",

  )

}

porcentaje <- function(column, compare, database) {
  registros <- nrow(database)
  coincidencias <-
    nrow(database %>% filter(database[[column]] == compare))
  value <- round(coincidencias / registros, 6)
  msg <- paste("$", value,
               "\\approx",
               value * 100, "\\%",
               "$", sep = "")
  resp <- list(msg = msg, value = value)
  return(resp)
}

binomial <- function (x, n, dataframe, column, compare) {
  registros <- dataframe %>% filter(dataframe[[column]] == compare)
  p <- round(nrow(registros) / nrow(dataframe), 6)

  fx <- cat(
    '<hr> $$ \\mbox{FUNCION DE PROBABILIDAD} $$ <br> $$ X \\sim B(' ,
    n ,
    ', ' ,
    p ,
    ') \\\\ \\begin{array}{lcc} f(x) = \\binom{',
    n ,
    '}{x}' ,
    p ,
    '^{x}(1-' ,
    p ,
    ')^{' ,
    n ,
    '-x} & si & \\quad x=0,1,2,3,... \\end{array} $$'
  )

  fxres <- cat(
    '$$ X \\sim B(',
    n ,
    ', ',
    p ,
    ') \\\\ \\begin{array}{lcc} f(',
    x ,
    ') = \\binom{',
    n ,
    '}{',
    x,
    '}' ,
    p ,
    '^{',
    x,
    '}(1-' ,
    p ,
    ')^{' ,
    n ,
    '-',
    x,
    '} & si & \\quad x=0,1,2,3,... \\end{array} $$ <br> $$ \\mbox{la probabilidad es del: }',
    dbinom(x, n, p),
    '\\approx',
    round(dbinom(x, n, p) * 100, 2),
    '\\% $$'
  )

  me <- cat(
    '<br> <hr> $$ \\mbox{MEDIA} $$ <br> $$ \\begin{align*} \\mu = E(X) & = np \\\\ & = ',
    n,
    '*',
    p,
    ' \\\\ & =  ',
    n * p,
    ' \\end{align*} $$'
  )

  dv <- cat(
    '<br> <hr> $$ \\mbox{VARIANZA} $$ <br> $$ \\begin{align*} \\sigma_x^2 & = np(1-p) \\\\ & = ',
    n,
    '*',
    p,
    '*(1-',
    p,
    ') \\\\ & =  ',
    (n * p) * (1 - p),
    ' \\end{align*} $$'
  )

  resp <- cat(fx, fxres, me)

  return(resp)
}
